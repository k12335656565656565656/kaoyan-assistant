# 用户画像系统 — 技术设计文档

> **目标**：让系统持续理解用户，拒绝每次从零开始
> **定位**：考研学习助手的核心数据层，支撑个性化推荐和智能提醒
> **技术栈**：Streamlit + SQLite + Z.ai GLM API

---

## 一、架构约束

本模块必须严格遵循现有架构，以下是不可更改的技术约束：

| 约束 | 说明 |
|------|------|
| **单文件架构** | 所有代码写入 `app.py`（当前 ~2100 行） |
| **前端框架** | Streamlit（不能用 React/Vue/HTML） |
| **数据库** | SQLite `data/memory.db`（不能用 MySQL/PostgreSQL） |
| **用户系统** | 复用 `st.session_state.user_id` 和 `username` |
| **导航** | `st.session_state.page` 状态机（值：`"profile"`） |
| **API** | Z.ai GLM（glm-4.6/flash/v-flash） |
| **样式** | 内联 CSS + Streamlit 组件（不能引入外部 CSS 框架） |
| **部署** | Ubuntu 服务器，`systemd` 管理，端口 8501 |

### 6.1 首次问卷建档（引导式多轮对话）

```python
def first_time_questionnaire(user_id):
    """首次登录引导问卷"""
    st.markdown("## 👋 欢迎！让我们了解一下你")
    
    if "questionnaire_step" not in st.session_state:
        st.session_state.questionnaire_step = 1
    
    step = st.session_state.questionnaire_step
    
    if step == 1:
        with st.form("step1"):
            st.markdown("### 第 1 步：基本信息")
            grade = st.selectbox("年级", ["大一", "大二", "大三", "大四", "往届"])
            major = st.text_input("本科专业")
            school = st.text_input("本科院校")
            
            if st.form_submit_button("下一步"):
                save_profile_field(user_id, "grade", grade)
                save_profile_field(user_id, "major", major)
                save_profile_field(user_id, "undergraduate_school", school)
                st.session_state.questionnaire_step = 2
                st.rerun()
    
    elif step == 2:
        with st.form("step2"):
            st.markdown("### 第 2 步：报考信息")
            target_major = st.text_input("目标专业")
            target_region = st.text_input("目标地区")
            target_school = st.text_input("目标院校（冲刺）")
            
            if st.form_submit_button("下一步"):
                save_profile_field(user_id, "target_major", target_major)
                save_profile_field(user_id, "target_region", target_region)
                save_profile_field(user_id, "target_schools", json.dumps({"冲刺": target_school}))
                st.session_state.questionnaire_step = 3
                st.rerun()
    
    elif step == 3:
        with st.form("step3"):
            st.markdown("### 第 3 步：学习状态")
            daily_hours = st.slider("每日可投入时长（小时）", 1, 12, 6)
            math_type = st.selectbox("数学考试类型", ["数一", "数二", "数三", "不考", "199管综"])
            weak_subject = st.text_input("弱科（如：政治）")
            
            if st.form_submit_button("完成建档"):
                save_profile_field(user_id, "daily_hours", daily_hours)
                save_profile_field(user_id, "math_exam_type", math_type)
                save_profile_field(user_id, "weak_subjects", json.dumps([weak_subject]))
                st.session_state.questionnaire_step = 0  # 完成
                st.success("🎉 画像建档完成！")
                st.rerun()
```

### 6.3 动态更新逻辑

```python
def update_profile_from_conversation(user_id, query, answer):
    """从对话中自动提取信息更新画像"""
    
    # 使用 LLM 提取信息
    prompt = f"""从以下对话中提取用户信息，返回 JSON 格式：
    
对话：
用户：{query}
AI：{answer}

请提取以下信息（如果有的话）：
- target_major: 目标专业
- target_school: 目标院校
- math_type: 数学类型（数一/数二/数三）
- weak_subject: 弱科
- anxiety_level: 焦虑程度（1-5）
- current_phase: 当前阶段（基础/强化/冲刺）

只返回 JSON，不要其他内容。如果没有信息，返回空 JSON {{}}"""
    
    try:
        result = call_llm_api(prompt, model="glm-4-flash")
        extracted = json.loads(result)
        
        for key, value in extracted.items():
            if value:
                save_profile_field(user_id, key, value)
                
    except Exception as e:
        pass  # 静默失败，不影响用户体验
```

### 6.4 标签自动生成

```python
def auto_generate_tags(user_id):
    """根据画像自动生成标签"""
    profile = get_user_profile(user_id)
    
    tags = {
        "common_errors": [],
        "strong_areas": [],
        "current_phase": profile.get("current_phase", "基础")
    }
    
    # 根据弱科生成常见错误标签
    weak_subjects = json.loads(profile.get("weak_subjects", "[]"))
    if "数学" in weak_subjects:
        tags["common_errors"].extend(["计算错误", "公式混淆"])
    if "英语" in weak_subjects:
        tags["common_errors"].extend(["语法错误", "词汇量不足"])
    
    # 根据强科生成优势领域标签
    strong_subjects = json.loads(profile.get("strong_subjects", "[]"))
    if "数学" in strong_subjects:
        tags["strong_areas"].extend(["逻辑推理", "公式应用"])
    
    # 保存标签
    save_profile_field(user_id, "common_errors", json.dumps(tags["common_errors"]))
    save_profile_field(user_id, "strong_areas", json.dumps(tags["strong_areas"]))
    save_profile_field(user_id, "current_phase", tags["current_phase"])
    
    return tags
```

### 6.5 内容安全检查

```python
def check_content_safety(answer, context):
    """检查内容安全性"""
    violations = []
    
    # 1. 检查是否虚构数据
    if re.search(r'(报录比|招生人数|分数线).*\d+', answer):
        if "来源" not in answer and "数据来源" not in answer:
            violations.append("数据未标注来源")
    
    # 2. 检查是否承诺录取
    if re.search(r'(保证|承诺|一定).*(上岸|录取|通过)', answer):
        violations.append("禁止承诺录取")
    
    # 3. 检查是否未经验证的政策解读
    if re.search(r'(政策|规定|要求).*解读', answer):
        if "建议核实" not in answer and "官方" not in answer:
            violations.append("政策解读未标注核实建议")
    
    # 4. 强制 RAG 检索高风险问题
    high_risk_keywords = ["分数线", "报录比", "招生人数", "政策变化"]
    if any(keyword in context.get("query", "") for keyword in high_risk_keywords):
        if not context.get("rag_used"):
            violations.append("高风险问题未使用 RAG 检索")
    
    return violations
```

## 七、辅助函数

```python
def get_user_profile(user_id):
    """获取用户画像"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("SELECT * FROM user_profiles WHERE user_id=?", (user_id,))
    row = c.fetchone()
    conn.close()
    
    if not row:
        return {}
    
    # 获取列名
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("PRAGMA table_info(user_profiles)")
    columns = [col[1] for col in c.fetchall()]
    conn.close()
    
    return dict(zip(columns, row))

def save_profile_field(user_id, field, value):
    """保存单个画像字段"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    
    # 检查是否存在
    c.execute("SELECT id FROM user_profiles WHERE user_id=?", (user_id,))
    exists = c.fetchone()
    
    if exists:
        c.execute(f"UPDATE user_profiles SET {field}=?, updated_at=CURRENT_TIMESTAMP WHERE user_id=?", 
                  (value, user_id))
    else:
        c.execute(f"INSERT INTO user_profiles (user_id, {field}) VALUES (?, ?)", 
                  (user_id, value))
    
    conn.commit()
    conn.close()

def get_consecutive_days(user_id):
    """计算连续打卡天数"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("""SELECT checkin_date FROM daily_checkin 
                 WHERE user_id=? ORDER BY checkin_date DESC""", (user_id,))
    dates = [row[0] for row in c.fetchall()]
    conn.close()
    
    if not dates:
        return 0
    
    streak = 0
    today = datetime.now().date()
    for d in dates:
        expected = (today - __import__('datetime').timedelta(days=streak)).strftime("%Y-%m-%d")
        if d == expected:
            streak += 1
        else:
            break
    return streak

def call_llm_api(prompt, model="glm-4-flash", max_tokens=500):
    """调用 LLM API"""
    data = {
        "model": model,
        "messages": [{"role": "user", "content": prompt}],
        "max_tokens": max_tokens,
        "temperature": 0.3
    }
    req = urllib.request.Request(
        API_BASE + "/chat/completions",
        data=json.dumps(data).encode("utf-8"),
        headers={"Content-Type": "application/json", "Authorization": f"Bearer {API_KEY}"},
        method="POST"
    )
    with urllib.request.urlopen(req, timeout=30) as resp:
        return json.loads(resp.read().decode("utf-8"))["choices"][0]["message"]["content"]
```
