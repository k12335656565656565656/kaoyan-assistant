# 学习规划模块 — 技术设计文档

> **目标**：生成并动态调整年/月/周/日计划
> **定位**：考研学习助手的核心规划功能
> **技术栈**：Streamlit + SQLite + Z.ai GLM API
> **架构**：规则引擎 + LLM 混合架构

---

## 一、架构约束

本模块必须严格遵循现有架构，以下是不可更改的技术约束：
不能引入任何新的 pip 依赖（除 Streamlit 外）
不能修改现有的数据库表结构
不能修改现有的页面路由逻辑
所有 SQL 操作必须使用 user_id 过滤（多用户隔离）
所有时间字段使用 YYYY-MM-DD 格式
规则引擎处理确定性计算，LLM 只负责个性化表述

| 约束 | 说明 |
|------|------|
| **单文件架构** | 所有代码写入 `app.py`（当前 ~2100 行） |
| **前端框架** | Streamlit（不能用 React/Vue/HTML） |
| **数据库** | SQLite `data/memory.db`（不能用 MySQL/PostgreSQL） |
| **用户系统** | 复用 `st.session_state.user_id` 和 `username` |
| **导航** | `st.session_state.page` 状态机（值：`"planner"`） |
| **API** | Z.ai GLM（glm-4.6/flash） |
| **样式** | 内联 CSS + Streamlit 组件 |
| **部署** | Ubuntu 服务器，`systemd` 管理，端口 8501 |

## 二、架构设计：规则引擎 + LLM 混合

### 设计原则

| 职责 | 实现 | 原因 |
|------|------|------|
| **确定性计算** | 规则引擎（Python） | 科目权重、时间分配、阶段判定、进度偏差 |
| **个性化表述** | LLM（glm-4-flash） | 计划描述、建议生成、激励文案 |
| **数据持久化** | SQLite | 计划存储、进度记录、历史回溯 |

### 数据流

```
用户输入（目标院校/考试类型/每日时长）
    ↓
规则引擎 → 计算科目权重 + 阶段划分 + 时间分配
    ↓
LLM → 生成个性化计划描述 + 学习建议
    ↓
SQLite → 存储计划 + 任务 + 进度
    ↓
Streamlit → 展示计划 + 进度追踪
```

### 阶段判定函数

```python
def determine_phase():
    """根据当前日期自动判定学习阶段"""
    month = datetime.now().month
    if 3 <= month <= 6:
        return "基础阶段"
    elif 7 <= month <= 9:
        return "强化阶段"
    elif 10 <= month <= 11:
        return "提升阶段"
    elif month == 12:
        return "冲刺阶段"
    else:
        return "基础阶段"
```

## 五、科目权重分配

### 权重表

| 科目 | 数一 | 数二 | 数三 | 不考数学 |  |
|------|------|------|------|----------|---------|
| 数学 | 35% | 35% | 35% | — | 
| 英语 | 20% | 20% | 20% | 30% | 
| 政治 | 15% | 15% | 15% | 20% | 
| 专业课 | 30% | 30% | 30% | 50% | 

### 权重计算函数

```python
def get_subject_weights(math_type):
    """根据数学考试类型返回科目权重"""
    weights = {
        "数一": {"数学": 0.35, "英语": 0.20, "政治": 0.15, "专业课": 0.30},
        "数二": {"数学": 0.35, "英语": 0.20, "政治": 0.15, "专业课": 0.30},
        "数三": {"数学": 0.35, "英语": 0.20, "政治": 0.15, "专业课": 0.30},
        "不考数学": {"英语": 0.30, "政治": 0.20, "专业课": 0.50},
      
    return weights.get(math_type, weights["数一"])

def calculate_daily_hours(daily_hours, math_type):
    """计算各科目每日学习时长"""
    weights = get_subject_weights(math_type)
    return {subject: round(daily_hours * weight, 1) for subject, weight in weights.items()}
```

### 补救策略选择逻辑

```python
def select_recovery_strategy(deviation, recent_completion_rates):
    """根据偏差程度和近期完成率选择补救策略"""
    strategies = []
    
    if deviation > 40:
        strategies.append(4)  # 切换速通模式
    if deviation > 30:
        strategies.append(2)  # 聚焦高频核心
    if deviation > 25:
        strategies.append(3)  # 压缩次要科目
    if deviation > 20:
        strategies.append(1)  # 删减低频考点
    
    if len(recent_completion_rates) >= 3:
        avg_rate = sum(recent_completion_rates[:3]) / 3
        if avg_rate < 60:
            strategies.append(5)  # 错题优先
        if len(recent_completion_rates) >= 5:
            avg_rate_5 = sum(recent_completion_rates[:5]) / 5
            if avg_rate_5 < 70:
                strategies.append(6)  # 调整作息
    
    return strategies
```

## 九、核心函数

### 9.1 计划生成

```python
def generate_plan(user_id, target_date, math_type, daily_hours):
    """生成完整学习计划"""
    weights = get_subject_weights(math_type)
    phase = determine_phase()
    daily分配 = calculate_daily_hours(daily_hours, math_type)
    
    days_remaining = (target_date - datetime.now().date()).days
    
    tasks = []
    for subject, weight in weights.items():
        subject_hours = daily_hours * weight
        phase_tasks = PHASE_TEMPLATES[phase].get(subject, [])
        
        for task_name in phase_tasks:
            tasks.append({
                "subject": subject,
                "task_name": task_name,
                "estimated_hours": subject_hours / len(phase_tasks),
                "priority": 3
            })
    
    prompt = f"""你是考研学习规划专家。根据以下信息生成个性化学习计划描述：

目标考试日期：{target_date}
剩余天数：{days_remaining}天
数学类型：{math_type}
每日学习时长：{daily_hours}小时
当前阶段：{phase}
各科权重：{json.dumps(weights, ensure_ascii=False)}

请生成：
1. 年度路线图概述
2. 当前阶段核心目标
3. 本周重点任务
4. 学习建议

要求简洁明了，适合考生快速阅读。"""
    
    description = call_llm_api(prompt, model="glm-4-flash")
    
    return {
        "description": description,
        "tasks": tasks,
        "phase": phase,
        "weights": weights,
        "daily分配": daily分配
    }
```

### 9.2 进度计算

```python
def calculate_progress(user_id):
    """计算学习进度"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    
    c.execute("""SELECT 
        COUNT(*) as total,
        SUM(CASE WHEN status='completed' THEN 1 ELSE 0 END) as completed
        FROM plan_tasks WHERE user_id=?""", (user_id,))
    total, completed = c.fetchone()
    
    c.execute("""SELECT subject,
        COUNT(*) as total,
        SUM(CASE WHEN status='completed' THEN 1 ELSE 0 END) as completed
        FROM plan_tasks WHERE user_id=? GROUP BY subject""", (user_id,))
    subjects = {}
    for row in c.fetchall():
        subjects[row[0]] = {
            "total": row[1],
            "completed": row[2],
            "completion_rate": round(row[2] / max(row[1], 1) * 100, 1)
        }
    
    c.execute("""SELECT planned_hours, actual_hours 
        FROM plan_progress WHERE user_id=? ORDER BY record_date DESC LIMIT 7""", (user_id,))
    recent = c.fetchall()
    deviation = 0
    if recent:
        planned = sum(r[0] for r in recent)
        actual = sum(r[1] for r in recent)
        deviation = round((planned - actual) / max(planned, 1) * 100, 1)
    
    conn.close()
    
    return {
        "total_tasks": total,
        "completed_tasks": completed,
        "completion_rate": round(completed / max(total, 1) * 100, 1),
        "deviation": deviation,
        "subjects": subjects
    }
```

def execute_recovery_strategy(user_id, strategy_id):
    """执行补救策略"""
    strategy = RECOVERY_STRATEGIES[strategy_id]
    
    if strategy_id == 1:  # 删减低频考点
        conn = sqlite3.connect(MEMORY_DB)
        c = conn.cursor()
        c.execute("""UPDATE plan_tasks SET status='skipped' 
                     WHERE user_id=? AND priority >= 4 AND status='pending'""",
                  (user_id,))
        conn.commit()
        conn.close()
    
    elif strategy_id == 2:  # 聚焦高频核心
        conn = sqlite3.connect(MEMORY_DB)
        c = conn.cursor()
        c.execute("""UPDATE plan_tasks SET priority=1 
                     WHERE user_id=? AND status='pending'""",
                  (user_id,))
        conn.commit()
        conn.close()
    
    elif strategy_id == 5:  # 错题优先
        conn = sqlite3.connect(MEMORY_DB)
        c = conn.cursor()
        c.execute("""SELECT knowledge_id FROM knowledge_mastery 
                     WHERE user_id=? AND times_wrong > 0 
                     ORDER BY times_wrong DESC LIMIT 5""", (user_id,))
        for row in c.fetchall():
            c.execute("""INSERT INTO plan_tasks 
                        (user_id, task_type, subject, task_name, description, 
                         target_date, estimated_hours, priority)
                        VALUES (?, 'daily', '数学', '错题复习', ?, 
                                date('now'), 1.0, 1)""",
                     (user_id, f"复习错题：{row[0]}"))
        conn.commit()
        conn.close()
    
    log_visit("执行补救策略", f"策略{strategy_id}: {strategy['name']}")
