# 打卡与督学模块 — 技术设计文档

> **目标**：提升学习执行率和用户留存
> **定位**：考研学习助手的第 5 个功能模块
> **技术栈**：Streamlit + SQLite + Z.ai GLM API

---

## 一、架构约束

本模块必须严格遵循现有架构，以下是不可更改的技术约束：

| 约束 | 说明 |
|------|------|
| **单文件架构** | 所有代码写入 `app.py`（当前 ~2100 行） |
| **前端框架** | Streamlit（不能用 React/Vue/HTML） |
| **数据库** | SQLite `data/memory.db`（不能用 MySQL/PostgreSQL） |
| **用户系统** | 复用 `st.session_state.user_id` 和 `username` |
| **导航** | `st.session_state.page` 状态机（值：`"checkin"`） |
| **API** | Z.ai GLM（glm-4.6/flash/v-flash） |
| **样式** | 内联 CSS + Streamlit 组件（不能引入外部 CSS 框架） |
| **部署** | Ubuntu 服务器，`systemd` 管理，端口 8501 |

# 每日打卡记录
c.execute("""CREATE TABLE IF NOT EXISTS daily_checkin (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    checkin_date TEXT NOT NULL,
    subject TEXT,
    duration_minutes INTEGER,
    completion_rate REAL,
    mood TEXT,
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    completed INTEGER DEFAULT 0,

    id INTEGER PRIMARY KEY AUTOINCREMENT,

    tasks TEXT,

    status TEXT DEFAULT 'active',

## 七、智能提醒系统

```python
def check_reminders(user_id):
    """检查并生成提醒列表"""
    reminders = []
    today = datetime.now()
    
    # 1. 连续3天完成率<60% → 警告
    recent = get_recent_checkins(user_id, days=3)
    if len(recent) >= 3 and all(r['completion_rate'] < 60 for r in recent[:3]):
        reminders.append(("warning", "⚠️ 连续3天完成率低于60%，建议调整学习计划"))
    
    # 2. 连续7天未打卡 → 掉队提醒
    last = get_last_checkin_date(user_id)
    if last:
        days_gap = (today - datetime.strptime(last, "%Y-%m-%d")).days
        if days_gap >= 7:
            reminders.append(("error", f"🚨 已经{days_gap}天没有打卡了，赶紧回来学习！"))
    
    # 3. 连续打卡里程碑
    streak = get_consecutive_days(user_id)
    if streak in [7, 21, 50, 100]:
        reminders.append(("success", f"🎉 恭喜连续打卡{streak}天！继续保持！"))
    
    # 4. 计划进度落后20%
    plan_progress = get_plan_progress(user_id)
    if plan_progress < 80 and plan_progress > 0:
        reminders.append(("warning", f"📊 当前计划进度{plan_progress}%，落后于预期，建议调整"))
    
    return reminders

# 在页面顶部显示提醒
reminders = check_reminders(user_id)
for level, msg in reminders:
    if level == "warning":
        st.warning(msg)
    elif level == "error":
        st.error(msg)
    elif level == "success":
        st.success(msg)
```

## 八、辅助函数

```python
def get_consecutive_days(user_id):
    """计算连续打卡天数"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("""SELECT checkin_date FROM daily_checkin 
                 WHERE user_id=? ORDER BY checkin_date DESC""", (user_id,))
    dates = [row[0] for row in c.fetchall()]
    conn.close()
    
    if not dates:
        return 0
    
    streak = 0
    today = datetime.now().date()
    for d in dates:
        expected = (today - __import__('datetime').timedelta(days=streak)).strftime("%Y-%m-%d")
        if d == expected:
            streak += 1
        else:
            break
    return streak

def get_today_duration(user_id):
    """获取今日学习时长"""
    today = datetime.now().strftime("%Y-%m-%d")
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("SELECT duration_minutes FROM daily_checkin WHERE user_id=? AND checkin_date=?", 
              (user_id, today))
    row = c.fetchone()
    conn.close()
    return row[0] if row else 0

def get_plan_progress(user_id):
    """获取当前计划平均进度"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("SELECT progress FROM study_plans WHERE user_id=? AND status='active'", (user_id,))
    rows = c.fetchall()
    conn.close()
    if not rows:
        return 0
    return round(sum(r[0] for r in rows) / len(rows))

def is_checked_today(user_id, today):
    """检查今天是否已打卡"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("SELECT id FROM daily_checkin WHERE user_id=? AND checkin_date=?", (user_id, today))
    result = c.fetchone()
    conn.close()
    return result is not None

def save_checkin(user_id, date, subject, duration, completion, mood, notes):
    """保存打卡记录"""
    conn = sqlite3.connect(MEMORY_DB)
    c = conn.cursor()
    c.execute("""INSERT OR REPLACE INTO daily_checkin 
                 (user_id, checkin_date, subject, duration_minutes, completion_rate, mood, notes)
                 VALUES (?, ?, ?, ?, ?, ?, ?)""",
              (user_id, date, subject, duration, completion, mood, notes))
    conn.commit()
    conn.close()
    log_visit("打卡", f"{subject} {duration}分钟 {completion}%")
```
